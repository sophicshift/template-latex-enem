% Circula alguma letra
\newcommand*{\circled}[1]{%
  \tikz[baseline=(char.base)]{
    \node[
      shape=circle,
      fill,
      text=white,
      font=\bfseries,
      inner sep=0.5pt
    ] (char) {#1};
  }
}

% Alternativas

    leftmargin=8mm,
    itemsep=-3pt,
    topsep=1ex
    ]
  }{%
  \end{enumerate}
}
